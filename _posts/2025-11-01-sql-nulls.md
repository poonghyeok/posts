#### MYSQL에서 NULL처리
```sql
select ifnull(null, 'a'); -- a
select coalesce(null, 'b', null, 'A'); -- b
select nullif(0, 0) -- null ... null을 처리를 하는 것이 아니라 null을 나와게 만들 수 있다.

select not(0 = null); -- null ... null은 말 그대로 존재하지 않는 것인데 비교한다는 것이 말이 안됨. 
select not(0 <=> null); -- true ... null safe 연산을 사용하면 됨.
```

#### ORACLE 19c에서 NULL처리
```sql
select NVL(null, 'a') FROM dual; -- a
select NVL2(123, 'a', 'b') FROM dual; -- a
select NVL2(null, 'a', 'b') FROM dual; -- b
select coalesce(null, 'b', null, 'A') FROM dual; -- b
select nullif(0, 0) FROM dual -- null ... null을 처리를 하는 것이 아니라 null을 나와게 만들 수 있다.  
```

> `NOT NULL`, `NOT IN` 과 같은 것은 사용할 수 있는데, `NOT(1=2)`는 사용할 수 없다.
